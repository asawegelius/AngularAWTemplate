import { HttpClient } from '@angular/common/http';
import { Injectable } from '@angular/core';
import { catchError, map } from 'rxjs/operators';
import { Endpoints } from 'src/app/modules/core/constants/endpoints';
import { CreateUrlService } from 'src/app/modules/core/services/create-url.service';
import { ResourceService } from 'src/app/modules/core/services/resource.service';
import { QueryStringParameters } from 'src/app/modules/core/utils/query-string-parameters';
import { IRoomReservations } from '../models/room-reservations.interface';

@Injectable({
  providedIn: 'root'
})
export class RoomReservationsApiService extends ResourceService<IRoomReservations> {
  getResourceUrl(): string {
    return this.urls.createUrl(this.endpoints.RESERVATIONS);
  }

  constructor(
    protected httpClient: HttpClient,
    private endpoints: Endpoints,
    private urls: CreateUrlService) {
    super(httpClient);
  }


  getByDate(date: Date) {
    const url = this.urls.createUrlWithQueryParameters(this.endpoints.RESERVATIONS, (qs: QueryStringParameters) => {
      qs.push('date', date);
    }, false);
    return this.httpClient.get<IRoomReservations>(url)
      .pipe(
        map((json) => this.fromServerModel(json)),
        catchError(this.handleError)
      );
  }
}
